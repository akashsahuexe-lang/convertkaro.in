<script>
let originalFileName = "document";

// Drag & Drop Setup
const zone = document.getElementById('uploadZone');
const fileInput = document.getElementById('fileInput');

zone.addEventListener('dragover', e => { e.preventDefault(); zone.classList.add('drag-on'); });
zone.addEventListener('dragleave', () => zone.classList.remove('drag-on'));
zone.addEventListener('drop', e => { e.preventDefault(); zone.classList.remove('drag-on'); handleFile(e.dataTransfer.files[0]); });
fileInput.addEventListener('change', e => handleFile(e.target.files[0]));

function handleFile(file) {
    if (!file) return;
    
    // Check if it's a docx file
    if (!file.name.endsWith('.docx')) {
        alert("Please upload a valid .docx Word file.");
        return;
    }

    originalFileName = file.name.replace('.docx', '');
    document.getElementById('fileName').textContent = file.name;
    document.getElementById('fileSize').textContent = (file.size / 1024).toFixed(1) + ' KB';

    document.getElementById('uploadZone').style.display = 'none';
    document.getElementById('loadingMsg').style.display = 'block';

    // Read the file and parse with Mammoth.js
    const reader = new FileReader();
    reader.onload = function(event) {
        const arrayBuffer = event.target.result;
        
        mammoth.convertToHtml({arrayBuffer: arrayBuffer})
            .then(displayResult)
            .catch(handleError);
    };
    reader.readAsArrayBuffer(file);
}

function displayResult(result) {
    // Hide loader, show workspace
    document.getElementById('loadingMsg').style.display = 'none';
    document.getElementById('workspace').style.display = 'flex';
    
    // Inject the HTML into our A4 preview div
    document.getElementById('docPreview').innerHTML = result.value;
    
    // Check if there are warnings (like unsupported complex formatting)
    if (result.messages && result.messages.length > 0) {
        console.warn("Mammoth messages:", result.messages);
    }
}

function handleError(err) {
    console.error(err);
    alert("Error reading the Word document. Make sure it's a valid .docx file.");
    location.reload();
}

// Convert the HTML Preview into a PDF using html2pdf.js
function downloadPDF() {
    const btn = document.getElementById('btnDownload');
    btn.innerText = "⏳ Generating PDF...";
    btn.disabled = true;

    const element = document.getElementById('docPreview');
    
    // html2pdf options for high quality A4 PDF
    const opt = {
        margin:       [10, 10, 10, 10], // top, left, bottom, right in mm
        filename:     `${originalFileName}_convertkaro.pdf`,
        image:        { type: 'jpeg', quality: 0.98 },
        html2canvas:  { scale: 2, useCORS: true },
        jsPDF:        { unit: 'mm', format: 'a4', orientation: 'portrait' }
    };

    // Generate and download
    html2pdf().set(opt).from(element).save().then(() => {
        btn.innerText = "✅ Downloaded Successfully";
        setTimeout(() => {
            btn.innerText = "⬇ Download as PDF";
            btn.disabled = false;
        }, 3000);
    });
}
